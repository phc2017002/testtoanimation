\begin{align*}
\text{def heapify\_down}(A, i):\\ n = \text{len}(A)\\ \text{largest} = i\\ l = 2*i + 1; \ r = 2*i + 2\\ \text{if } l < n \text{ and } A[l] > A[\text{largest}]: \text{largest} = l\\ \text{if } r < n \text{ and } A[r] > A[\text{largest}]: \text{largest} = r\\ \text{if largest} \neq i:\\ \qquad \text{swap}(A,i,\text{largest})\\ \qquad \text{heapify\_down}(A, \text{largest})
\end{align*}
