\begin{align*}
\text{def heapify-up}(A, i):\\ \text{while } i > 0:\\ \quad p = \text{floor}((i-1)/2)\\ \quad \text{if } A[i] > A[p]:\\ \quad \quad \text{swap}(A,i,p)\\ \quad \quad i = p\\ \quad \text{else: break}
\end{align*}
