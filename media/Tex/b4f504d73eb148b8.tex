\begin{align*}
\text{def heapify-down}(A, i):\\ n = \text{len}(A)\\ largest = i\\ l = 2*i + 1; r = 2*i + 2\\ \text{if } l < n \text{ and } A[l] > A[largest]: largest = l\\ \text{if } r < n \text{ and } A[r] > A[largest]: largest = r\\ \text{if } largest != i:\\ \quad \text{swap}(A,i,largest)\\ \quad \text{heapify-down}(A, largest)
\end{align*}
