\begin{align*}
\text{def heapify\_up}(A, i):\\ \text{while } i > 0:\\ \qquad p = \lfloor(i-1)/2\rfloor\\ \qquad \text{if } A[i] > A[p]:\\ \qquad \qquad \text{swap}(A,i,p)\\ \qquad \qquad i = p\\ \qquad \text{else: break}
\end{align*}
